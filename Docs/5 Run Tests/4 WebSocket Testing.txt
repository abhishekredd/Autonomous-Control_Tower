# Install WebSocket client
pip install websocket-client

# Test WebSocket connection using Python
python -c "
import websocket
import json

ws = websocket.WebSocket()
ws.connect('ws://localhost:8000/ws/shipments/1')

# Send a test message
ws.send(json.dumps({'type': 'subscribe', 'shipment_id': 1}))

# Receive messages
for i in range(5):
    result = ws.recv()
    print(f'Message {i+1}: {result}')

ws.close()
"
