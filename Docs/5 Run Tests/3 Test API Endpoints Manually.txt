# 1. Create a shipment
curl -X POST "http://localhost:8000/api/v1/shipments/" \
  -H "Content-Type: application/json" \
  -d '{
    "tracking_number": "TEST123456",
    "reference_number": "TEST-SH-001",
    "origin": "Shanghai, China",
    "destination": "Rotterdam, Netherlands",
    "mode": "sea",
    "weight": 25000.0,
    "volume": 68.0,
    "value": 500000.0,
    "estimated_departure": "2024-01-01T00:00:00Z",
    "estimated_arrival": "2024-01-15T00:00:00Z",
    "shipper": "Test Corporation",
    "carrier": "Test Shipping Line",
    "consignee": "Test Imports BV"
  }'

# 2. Get all shipments
curl "http://localhost:8000/api/v1/shipments/"

# 3. Get specific shipment
curl "http://localhost:8000/api/v1/shipments/1"

# 4. Trigger risk check
curl -X POST "http://localhost:8000/api/v1/shipments/1/trigger-risk-check"

# 5. Get risks for shipment
curl "http://localhost:8000/api/v1/risks/shipment/1"

# 6. Run simulation
curl -X POST "http://localhost:8000/api/v1/simulations/" \
  -H "Content-Type: application/json" \
  -d '{
    "shipment_id": 1,
    "simulation_type": "mitigation_analysis",
    "parameters": {"risk_type": "port_congestion"},
    "scenario_description": "Test simulation"
  }'
