# Create load test script (locustfile.py)
from locust import HttpUser, task, between

class ControlTowerUser(HttpUser):
    wait_time = between(1, 3)
    
    @task(3)
    def get_shipments(self):
        self.client.get("/api/v1/shipments/")
    
    @task(2)
    def get_risks(self):
        self.client.get("/api/v1/risks/")
    
    @task(1)
    def create_shipment(self):
        self.client.post("/api/v1/shipments/", json={
            "tracking_number": "LOAD-TEST",
            "origin": "Test",
            "destination": "Test",
            "mode": "sea"
        })

# Run load test
locust -f locustfile.py --host=http://localhost:8000
