# Backup database
docker-compose exec postgres pg_dump -U control_tower_user control_tower_db > backup_$(date +%Y%m%d).sql

# Restore database
docker-compose exec -T postgres psql -U control_tower_user control_tower_db < backup.sql

# Monitor system resources
docker stats

# Clean up Docker
docker system prune -af

# Update dependencies
docker-compose build --no-cache api
docker-compose up -d api

# View API metrics
curl http://localhost:8000/health
